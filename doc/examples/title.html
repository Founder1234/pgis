<div id="map" style="padding: 10px; border: 1px solid blue; width: 80%; height: 350px;"></div>
<script type="text/javascript">
	var map = new EzMap('map');

	/**
	 * 添加Title到地图
	 */
	var title1 = new EzTitle("我是一个孤独Title对象",{
		"fontSize": 16,
		"fontColor":"#FFFFFF",
		"fillColor": "#1215AB",
		"isStroke": true,
		"strokeColor": "#EE1E2F",
		"strokeWidth": 1,
		"lineHeight": 1.8,
		"paddingH": 4,
		"offset": [0,0]
	});
	map.addOverlay(title1);
	var position = new EzCoord(116.14765,39.94921);
	title1.setPosition(position);

	/**
	 * 通过marker添加Title到地图
	 */
	var title2 = new EzTitle("我是一个Title对象，和marker在一起",{
		"fontSize": 16,
		"fontColor":"#FFFFFF",
		"fillColor": "#1215AB",
		"isStroke": true,
		"strokeColor": "#EE1E2F",
		"strokeWidth": 1,
		"lineHeight": 2.5,
		"paddingH": 15,
		"offset": [0,-44]
	});
	var position2 = new EzCoord(116.64765,39.94921);
	var marker2 = new EzMarker(position2,undefined,title2);
	map.addOverlay(marker2);
	marker2.showTitle();
</script>
